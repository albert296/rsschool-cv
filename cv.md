```C#
public class Kata
{
  public static string DuplicateEncode(string word)
  {
    word= word.ToLower();
    var arr = new string[word.Length];
    for (int i = 0; i < word.Length; i++)
    {
        var duplicateCount = 0;
        foreach (var t in word)
        {
            if (word[i] == t)
                duplicateCount++;
        }

        if (duplicateCount >= 2) arr[i] = ")";
        else arr[i] = "(";
    }
    return string.Join("", arr);
  }
}
```
